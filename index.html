<script>


const body=document.getElementById("tableBody");


function createRow(i){


const tr=document.createElement("tr");


tr.innerHTML=`

<td>${i}</td>

<td><input type="text"></td>

<td><input type="datetime-local"></td>

<td>

<select>

<option>To Do</option>

<option>Planned</option>

<option>Done</option>

</select>

</td>

`;


const text=tr.querySelector("input[type=text]");

const date=tr.querySelector("input[type=datetime-local]");

const status=tr.querySelector("select");


text.addEventListener("input",save);

date.addEventListener("change",save);


status.addEventListener("change",()=>{


if(status.value==="Done"){

confetti({

particleCount:120,

spread:90,

origin:{y:.7}

});

}


updateProgress();

save();


});


body.appendChild(tr);


}


for(let i=1;i<=100;i++){

createRow(i);

}


function updateProgress(){


const rows=document.querySelectorAll("tbody tr");

let done=0;


rows.forEach(r=>{

const s=r.querySelector("select").value;


if(s==="Done"){

done++;

r.classList.add("done");

}else{

r.classList.remove("done");

}

});


document.getElementById("counter").innerText =

done+" / 100 completate";


document.getElementById("percent").innerText =

done+"%";


document.getElementById("progressBar").style.width =

done+"%";

}


function save(){


const data=[];


document.querySelectorAll("tbody tr").forEach(r=>{


data.push({

activity:r.querySelector("input[type=text]").value,

date:r.querySelector("input[type=datetime-local]").value,

status:r.querySelector("select").value

});


});


localStorage.setItem("bucket100",JSON.stringify(data));


updateProgress();

}


function load(){


const data=JSON.parse(localStorage.getItem("bucket100"));

if(!data)return;


document.querySelectorAll("tbody tr").forEach((r,i)=>{


if(!data[i])return;


r.querySelector("input[type=text]").value=data[i].activity;

r.querySelector("input[type=datetime-local]").value=data[i].date;

r.querySelector("select").value=data[i].status;


});


updateProgress();

}


load();


</script>
